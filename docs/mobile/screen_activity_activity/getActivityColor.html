Color getActivityColor (ActivityType type)

Implementation 

Color getActivityColor(ActivityType type) {
  switch (type) {
    case ActivityType.CLIENT:
      return Colors.blue;
    case ActivityType.PROJECT:
      return Colors.pink  ;
    case ActivityType.RECEIPT:
      return Colors.amber;
    case ActivityType.STAGE:
      return Colors.red;
    case ActivityType.USER:
      return Colors.purple;
  }
  return null;
}
